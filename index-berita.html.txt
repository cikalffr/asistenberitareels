<script>
const API_KEY = ""; // Canvas will provide this
const API_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${API_KEY}`;

// DOM Elements
const newsInput = document.getElementById('newsInput');
const generateBtn = document.getElementById('generateBtn');
const clearBtn = document.getElementById('clearBtn');
const loadingState = document.getElementById('loadingState');
const resultContainer = document.getElementById('resultContainer');
const errorBox = document.getElementById('errorBox');
const errorMessage = document.getElementById('errorMessage');

// Result Elements
const slidesContainer = document.getElementById('slidesContainer');
const resHighlights = document.getElementById('resHighlights');
const resCaption = document.getElementById('resCaption');
const resHashtags = document.getElementById('resHashtags');
const resCaptionRaw = document.getElementById('resCaptionRaw');
const resAllRaw = document.getElementById('resAllRaw');
const resFormatBadge = document.getElementById('resFormatBadge');

// State Global untuk Regenerate
let currentInputData = "";
let currentSlideFormat = "3";

const outputSchema = {
type: "OBJECT",
properties: {
slides: {
type: "ARRAY",
description: "Daftar slide Canva yang dihasilkan. Jumlahnya HARUS sesuai dengan format yang diminta (1 atau 3).",
items: {
type: "OBJECT",
properties: {
slide_number: { type: "INTEGER", description: "Nomor urut slide (1, 2, atau 3)" },
judul_slide: { type: "STRING", description: "Konteks slide, contoh: 'Gambar/Slide 1 (Judul / Hook)' atau 'Gambar/Slide 2 (Penjelasan Kasus)'" },
background: { type: "STRING", description: "Saran visual/background spesifik. Contoh: 'Cocok dengan background foto tumpukan sampah yang menggunung...'" },
label_kuning: { type: "STRING", description: "Teks label topik (UPPERCASE). Contoh: 'PERISTIWA / LINGKUNGAN'" },
teks_utama: { type: "STRING", description: "Teks copywriting utama. Untuk slide 2 dan 3, buat 2-3 kalimat yang padat, jelas, dan terjelaskan dengan baik (tidak terlalu pendek)." }
},
required: ["slide_number", "judul_slide", "background", "label_kuning", "teks_utama"]
}
},
highlights: {
type: "ARRAY",
items: { type: "STRING" },
description: "Instruksi highlight biru. Contoh: 'Slide 1: Pada teks KEBAKARAN MELANDA TPA JATIWARINGIN'"
},
caption: { type: "STRING", description: "Caption Instagram/WA detail (2-3 paragraf) gaya santai jurnalistik dan Call to Action di akhir." },
hashtags: { type: "STRING", description: "Kumpulan hashtag (misal: #BeritaTerkini #Update)" }
},
required: ["slides", "highlights", "caption", "hashtags"]
};

clearBtn.addEventListener('click', () => {
newsInput.value = '';
hideError();
resultContainer.classList.add('hidden');
});

generateBtn.addEventListener('click', async () => {
const inputVal = newsInput.value.trim();
if (!inputVal) {
showError("Mohon masukkan link berita atau paste teks berita terlebih dahulu.");
return;
}

const slideFormat = document.querySelector('input[name="slideFormat"]:checked').value;

// Simpan state untuk keperluan regenerate
currentInputData = inputVal;
currentSlideFormat = slideFormat;

// UI State updates
hideError();
resultContainer.classList.add('hidden');
loadingState.classList.remove('hidden');
document.querySelector('#loadingState p').textContent = "AI sedang meracik berita menjadi format Canva...";
generateBtn.disabled = true;
generateBtn.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Meracik Konten...';

try {
const result = await generateKaynewsContent(inputVal, slideFormat, 'medium');
renderResults(result, slideFormat);
} catch (error) {
console.error("Error generating content:", error);
showError(error.message || "Terjadi kesalahan saat menghubungi AI. Silakan coba lagi.");
} finally {
loadingState.classList.add('hidden');
generateBtn.disabled = false;
generateBtn.innerHTML = '<i class="fa-solid fa-wand-magic-sparkles"></i> Buat Konten Kaynews';
}
});

// Fungsi Regenerate dengan panjang teks tertentu
async function regenerateContent(modifier) {
if (!currentInputData) return;

hideError();
resultContainer.classList.add('hidden');
loadingState.classList.remove('hidden');
document.querySelector('#loadingState p').textContent = `AI sedang menyesuaikan ulang panjang teks menjadi ${modifier === 'short' ? 'lebih singkat' : modifier === 'long' ? 'lebih detail' : 'standar'}...`;

try {
const result = await generateKaynewsContent(currentInputData, currentSlideFormat, modifier);
renderResults(result, currentSlideFormat);
} catch (error) {
console.error("Error regenerating content:", error);
showError(error.message || "Terjadi kesalahan saat menghubungi AI. Silakan coba lagi.");
} finally {
loadingState.classList.add('hidden');
}
}

async function generateKaynewsContent(input, format, textModifier = 'medium') {
let lengthInstruction = "";
if (textModifier === 'short') {
lengthInstruction = "PENTING: Buat teks utama pada slide 2 & 3 serta caption LEBIH SINGKAT, padat, dan to the point. Hindari kalimat bertele-tele.";
} else if (textModifier === 'long') {
lengthInstruction = "PENTING: Buat teks utama pada slide 2 & 3 serta caption LEBIH PANJANG, mendetail, dan komprehensif. Masukkan lebih banyak data, kutipan, atau konteks tambahan dari berita agar sangat informatif (Bisa 3-4 kalimat per slide).";
} else {
lengthInstruction = "Buat teks utama pada slide 2 & 3 dengan panjang STANDAR (2-3 kalimat utuh), padat, jelas, namun terjelaskan dengan baik.";
}

const prompt = `Anda adalah pembuat konten berita sosial media profesional untuk media bernama 'Kaynews'.
Tugas Anda adalah merangkum berita berikut dan meraciknya menjadi format teks untuk desain Carousel/Slide Canva (cocok untuk Instagram & WhatsApp).

Input Berita:
${input}

Instruksi Penting:
1. HARUS MENGHASILKAN TEPAT ${format} SLIDE.
2. Jika input adalah URL, gunakan pencarian (Google Search) untuk mengetahui isi beritanya.
3. Gaya bahasa: Jurnalistik namun santai, kekinian (engaging), dan informatif.
4. ${lengthInstruction}
5. Ketentuan Teks Utama Berdasarkan Format:
- JIKA 1 SLIDE: Fokus pada inti berita, memancing rasa penasaran, dan Judul/Teks Utama HARUS huruf besar semua (ALL CAPS).
- JIKA 3 SLIDE:
* Slide 1 (Judul/Hook): Judul berita yang sangat memancing perhatian (ALL CAPS).
* Slide 2 (Penjelasan Kasus/Isu): Berisi penjelasan kronologi, isu, atau akar permasalahan.
* Slide 3 (Dampak/Tindakan/Solusi): Berisi penjelasan dampak lanjutan, tindakan dari pihak berwenang, atau konklusi.
6. Caption: Buat caption yang relevan dengan panjang teks yang direquest, menceritakan ulang beritanya secara menarik, diakhiri dengan Call to Action (pertanyaan ke audiens) dan daftar hashtag.

Struktur yang Diharapkan:
- Judul Slide: Gunakan penamaan jelas. (Misal: Gambar/Slide 2 (Penjelasan Kasus)).
- Slide: Terdiri dari saran background visual, Label Kuning (Kategori), dan Teks Utama.
- Garis Biru (Highlights): Sebutkan kutipan teks mana saja di tiap slide yang perlu diberi highlight warna biru.
- Caption & Hashtags: Caption detail beserta hashtag yang relevan.`;

const payload = {
contents: [{ parts: [{ text: prompt }] }],
tools: [{ google_search: {} }],
generationConfig: {
responseMimeType: "application/json",
responseSchema: outputSchema,
temperature: 0.7
}
};

const response = await fetch(API_URL, {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(payload)
});

if (!response.ok) {
throw new Error(`HTTP error! status: ${response.status}`);
}

const result = await response.json();

if (result.candidates && result.candidates.length > 0 &&
result.candidates[0].content && result.candidates[0].content.parts &&
result.candidates[0].content.parts.length > 0) {

const jsonText = result.candidates[0].content.parts[0].text;
return JSON.parse(jsonText);
} else {
throw new Error("Format respons dari AI tidak sesuai atau kosong.");
}
}

function renderResults(data, format) {
resFormatBadge.textContent = `${format} Slide`;

// Clear previous slides
slidesContainer.innerHTML = '';
let rawAllText = `Teks untuk Slide Canva\n\n`;

// Populate Slides
data.slides.forEach(slide => {
const isTitleSlide = slide.slide_number === 1;

// Add to raw text for copying
rawAllText += `${slide.judul_slide}\n`;
rawAllText += `${slide.background}\n`;
rawAllText += `[Label Kuning]: ${slide.label_kuning}\n`;
rawAllText += `${slide.teks_utama}\n\n`;

// Build HTML Card
const card = document.createElement('div');
card.className = "bg-white rounded-2xl shadow-sm border border-gray-200 overflow-hidden";
card.innerHTML = `
<div class="bg-gray-100 border-b border-gray-200 px-4 py-2 flex justify-between items-center">
<span class="text-xs font-bold text-gray-500 uppercase">${slide.judul_slide}</span>
</div>
<div class="p-5">
<p class="text-sm text-gray-500 italic mb-4 flex gap-2">
<i class="fa-solid fa-image text-gray-400 mt-0.5"></i> ${slide.background}
</p>
<div class="bg-gray-50 border border-dashed border-gray-300 p-6 rounded-xl relative text-center flex flex-col items-center justify-center min-h-[150px]">
<div class="mb-3"><span class="label-kuning">${slide.label_kuning}</span></div>
<h3 class="${isTitleSlide ? 'text-2xl font-black uppercase text-gray-900 leading-tight' : 'text-lg font-medium text-gray-800'}">${slide.teks_utama}</h3>
</div>
</div>
`;
slidesContainer.appendChild(card);
});

// Populate Highlights
resHighlights.innerHTML = '';
rawAllText += `Rekomendasi "Garis Biru" (Highlight):\n`;
data.highlights.forEach(highlight => {
const li = document.createElement('li');
li.className = "flex items-start gap-2";
li.innerHTML = `<i class="fa-solid fa-check text-blue-500 mt-1"></i> <span class="font-medium">${highlight}</span>`;
resHighlights.appendChild(li);
rawAllText += `- ${highlight}\n`;
});
rawAllText += `\n`;

// Populate Caption
resCaption.textContent = data.caption;
resHashtags.textContent = data.hashtags;

// Raw text structures
resCaptionRaw.textContent = `${data.caption}\n\n${data.hashtags}`;

rawAllText += `Caption Instagram & WA (Versi Singkat)\n`;
rawAllText += `${data.caption}\n\n${data.hashtags}`;
resAllRaw.textContent = rawAllText;

// Show Container
resultContainer.classList.remove('hidden');
resultContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });
}

function copyText(elementId) {
executeCopy(document.getElementById(elementId).innerText || document.getElementById(elementId).textContent);
}

function copyAllText() {
executeCopy(document.getElementById('resAllRaw').textContent);
}

function executeCopy(textToCopy) {
const textArea = document.createElement("textarea");
textArea.value = textToCopy;
textArea.style.top = "0";
textArea.style.left = "0";
textArea.style.position = "fixed";
document.body.appendChild(textArea);
textArea.focus();
textArea.select();

try {
const successful = document.execCommand('copy');
if(successful) {
showTemporaryMessage("Teks berhasil disalin!");
} else {
showTemporaryMessage("Gagal menyalin teks.");
}
} catch (err) {
console.error('Oops, unable to copy', err);
}
document.body.removeChild(textArea);
}

function showTemporaryMessage(msg) {
const toast = document.createElement('div');
toast.className = "fixed bottom-5 left-1/2 transform -translate-x-1/2 bg-gray-800 text-white px-5 py-3 rounded-xl shadow-2xl text-sm font-medium z-50 transition-opacity duration-300 flex items-center gap-2";
toast.innerHTML = `<i class="fa-solid fa-circle-check text-green-400"></i> ${msg}`;
document.body.appendChild(toast);

setTimeout(() => {
toast.style.opacity = '0';
setTimeout(() => document.body.removeChild(toast), 300);
}, 2500);
}

function showError(msg) {
errorMessage.textContent = msg;
errorBox.classList.remove('hidden');
}

function hideError() {
errorBox.classList.add('hidden');
errorMessage.textContent = '';
}
